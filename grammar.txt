
expr :      un-expr
            expr bin-op expr

un-expr :   prim-expr
            un-op un-expr

prim-expr : operand

bin-op :     "||" | "&&" | rel_op | add_op | mul_op

rel_op :    "==" | "!=" | "<" | "<=" | ">" | ">="

add_op :    "+" | "-" | "|" | "^" 

mul_op :    "*" | "/" | "%" | "<<" | ">>" | "&" | "&^"

un-op :     "+" | "-" | "!" | "^" | "*" | "&"

operand :   op-name
            literal
            "(" expr ")"

op-name :   IDENTIFIER

literal :   INTEGER
            STRING
