prog : func


func : type IDENTIFIER () { block rtrn-stmt }
       type IDENTIFIER () { rtrn-stmt }
       IDENTIFIER () { block }

block : block stmt
        stmt

rtrn-stmt : RETURN stmt

stmt : expr ;
       builtinfunc ;

builtinfunc : PRINT ( expr )

expr : INT
     : IDENTIFIER
     : ( expr )
     : expr + expr
     : expr - expr
     : expr * expr
     : expr / expr


type : TYPE_INT

literals : INTEGER
